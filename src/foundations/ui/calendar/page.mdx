## API Reference

Extends the `div` element.

### Single

When `mode="single"{:tsx}` (or when ommited).

<PropsTable
  definition={{
    mode: {
      type: '"single"',
      default: '"single"',
    },
    value: {
      type: "Date",
      required: true,
    },
    onDateChange: {
      type: "(date: Date) => void",
      required: true,
    },
  }}
/>

### Range

When `mode="range"{:tsx}`.

<PropsTable
  definition={{
    mode: {
      type: '"range"',
      required: true,
    },
    value: {
      type: "[Date, Date]",
      required: true,
    },
    onDateChange: {
      type: "(dates: [Date, Date]) => void",
      required: true,
    },
  }}
/>

### Common

Applies to both `mode="single"{:tsx}` and `mode="range"{:tsx}`.

<PropsTable
  definition={{
    startWeekOn: {
      type: ["0", "1", "2", "3", "4", "5", "6"],
      default: "0",
    },
    locale: {
      type: "Intl.LocalesArgument",
      default: '"en"',
    },
    getIsDisabled: {
      type: "(date: Date) => boolean",
    },
  }}
/>
