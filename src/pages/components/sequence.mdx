# Sequence

A Sequence component that cycles through content based on a set interval.
Users can manually navigate through sections, or it will progress automatically when in view.

## Usage

```tsx copy
import { Sequence } from '[path]/Sequence';
```

```tsx copy
<Sequence.Root values={['Slide 1', 'Slide 2', 'Slide 3']} stepDuration={3000}>
  <div>
    <Sequence.Trigger value="Slide 1">Slide 1</Sequence.Trigger>
    <Sequence.Trigger value="Slide 2">Slide 2</Sequence.Trigger>
    <Sequence.Trigger value="Slide 3">Slide 3</Sequence.Trigger>
  </div>
  <div>
    <Sequence.Content value="Slide 1">Content for Slide 1</Sequence.Content>
    <Sequence.Content value="Slide 2">Content for Slide 2</Sequence.Content>
    <Sequence.Content value="Slide 3">Content for Slide 3</Sequence.Content>
  </div>
</Sequence.Root>
```
